# File lib/urwPriceCalc.rb, line 81
def process_command(cmd)
        args = Array(cmd)
        command = args.shift
        case(command)
        when "setup_db"
                URWPriceCalc::Models.generate_schema
        when "+q"
                addQuality(args[0], args[1]) rescue unreqCommand
        when "+p"
                addItemPrice(args[0], args[1]) rescue unreqCommand
        when "+i"
                addInventory(args[0], args[1]) rescue unreqCommand
        when "q"
                puts showQualities
        when "p"
                puts showItemPrices
        when "i"
                puts showInventory
        when "destroy"
                destroy
        else
                unreqCommand
        end
end
